`./index.html`
```html
<!DOCTYPE html>
<html>
  <head>
    <meta charset="utf-8">
    <title>test</title>
  </head>
  <body>
    <div id="app-desk">
      <!-- Here is app! -->
    </div>
    <script src="./dist/bundle.js" charset="utf-8"></script>
  </body>
</html>
```

`./app/component/App.vue`
```html
<template lang="html">
  <div class="app">
    <h1>{{ greeting }}</h1>
  </div>
</template>

<script>
export default {
  data() {
    return {
      greeting: 'Hi Hello'
    }
  }
}
</script>

<style lang="css">
.app {
  background-color: pink;
}
</style>
```

```js
import Vue from 'vue'
import App from './components/App.vue'

new Vue({
  el: '#app-desk',
  render: h => h(App)
})
```
